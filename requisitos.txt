#Xoniter Simple - Ejecutor de Comandos con Sudo
#Dependencias: flask, qrcode[pil], pillow

#Arch Linux
sudo pacman -S python-pip
pip install flask --break-system-packages
pip install qrcode[pil] --break-system-packages
pip install pillow --break-system-packages
#Ejecucion:
python app.py --host 0.0.0.0 --port 5100

#Ubuntu y derivados
sudo apt update
sudo apt install python3 python3-pip -y
pip3 install flask
pip3 install qrcode[pil]
pip3 install pillow
#Ejecucion:
python3 app.py --host 0.0.0.0 --port 5100

#Windows
pip install flask
pip install qrcode[pil]
pip install pillow
#Ejecucion:
python app.py --host 0.0.0.0 --port 5100

#macOS
brew install python3
pip3 install flask
pip3 install qrcode[pil]
pip3 install pillow
#Ejecucion:
python3 app.py --host 0.0.0.0 --port 5100

#Opciones adicionales:
# --no-sudo  : Ejecutar sin privilegios sudo
# --no-qr    : No mostrar codigo QR en terminal
# --timeout  : Tiempo limite para comandos (segundos)
# --port     : Puerto personalizado (defecto: 5100)

#Ejemplo con opciones:
python app.py --host 0.0.0.0 --port 5500 --timeout 30 --no-qr
